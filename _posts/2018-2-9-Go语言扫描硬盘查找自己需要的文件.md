### 写入文件保存

```go
func WriteFile(file []string){
	//保存文件名
	filename:="./output.txt"
	//判断文件是否存在，存在即删除
	_,err := os.Stat(filename)
	if err == nil{
		os.Remove(filename)
	}
	//打开此文件
	fd,_:=os.OpenFile(filename,os.O_RDWR|os.O_CREATE|os.O_APPEND,0644)
	//循环值并且转换类型
	for k,v:=range file{
		content:=strings.Join([]string{strconv.Itoa(k),"==> 查找路径:",v,"\n"},"")
		buf:=[]byte(content)
		fd.Write(buf)
	}
	defer fd.Close()
}
```
